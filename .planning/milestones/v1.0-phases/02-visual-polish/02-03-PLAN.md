```python
from streamlit_folium import st_folium
import folium

def display_map(latitude: float, longitude: float, location_name: str) -> None:
    """Display interactive map with city marker."""
    # Create map centered on location
    m = folium.Map(
        location=[latitude, longitude],
        zoom_start=10,
        tiles="CartoDB dark_matter"  # Dark theme tiles per locked decision
    )
    
    # Add circle marker (not standard pin) per locked decision
    folium.CircleMarker(
        location=[latitude, longitude],
        radius=10,
        color="#00d4ff",  # Cyan to match title gradient
        fill=True,
        fillColor="#00d4ff",
        fillOpacity=0.7,
        popup=location_name
    ).add_to(m)
    
    # Display in Streamlit
    st_folium(m, width=380, height=300)
```
Per CONTEXT: Use CartoDB dark_matter tiles, circle marker.
